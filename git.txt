//first check if you have installed git properly
git --version

//configure git to work with your name and email
//This is to track who made changes in the project
git config --global user.name "Your name"
git config --global user.email "Your email"

//Repository is where git tracks everything (all the versions of your codes) in your project
//for creating a local Repository
git init

//To make a default branch name created by git (i.e change it from master to main)
git config --global init.defaultBranch main

//Add 2 files (hello.js and readme.md) and type
git status
//git status : tells you the branch and many files that are not commited and that are untracked
//for this example, there are 2 untracked and uncommited files

//To add only a file
git add "name of the file"

//To add all files at once, plus folders (for example added a new file called test.js to show how to add all files at once instead of one file after the other)
git add .
//The " . " : means current directory

//To commit with a message (-m) is to track the file
git commit -m "write message of commit"

//Always run "git status" to see what your git is showing you
git status

//To see the history of all files commited
git log

//To restore to previous version (this is useful when the current version of code has a bug)
1.) copy the commit hash number
2.) write git checkout "paste the commit hash number"
3.) Notice the cancelled files. (no worries, the files are added and saved)

//To get back to the current version (from above the see the cancelled files again)
git checkout main

//To discard the cancelled files (from above i.e the restore section, if you do not want to see the cancelled files, )
git checkout -f main
// " -f " - means force discard

//To switch the master branch to main (this is a standard practice)
git branch -M main

//To link local Repository to remote Repository
git remote add origin https://github.com/mike-pj/mastering_git.git
// origin = https://github.com/mike-pj/mastering_git.git (origin - is the name of the url link)

//To push to github
git push -u origin main

******************************** BRANCHING AND MERGING **********************************
//To create a new branch (e.g branch-name name of new branch)
git branch branch-name

//To switch to the new branch created
git checkout branch-name

//To switch back to original branch main
git checkout main

//To create and move to a branch all at once (e.g feature-branch name of newly branch you created and want to enter)
git checkout -b feature-branch

//To switch back to original branch main
git checkout main

//To create and move to a branch all at once (e.g feature-branch name of newly branch you created and want to enter)
git checkout -b feature-branch

//To push a file into a newly switched branch e.g feature-branch (we go to our file "readme.md" modify, add, commit and push)
git push --set-upstream origin feature-branch
git push -u origin feature-branch
//both codes can be used above

//If anyone made modifications or changes in any of your remote Repository branches, we have to make those changes merge and appear in our local Repository. like make it sync.
git pull

//A "pull request" let's you share your changes with your team for review and feed back. Once approved and merged, your changes become a part of the main branch, keeping the code base stable and organized
on your github:
1.) Click on pull request
2.) Click on new request
3.) Click on branch you want to merge from (in our own case, feature-branch)
4.) Click on merge

//To make all above appear on your local Repository (appear on your computer)
git pull 
git pull -u origin main (another way to write it)
